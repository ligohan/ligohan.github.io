2018-05-11

关于DSL

  • es中的查询请求有两种方式:
    • 一种是简易版的查询，像传递URL参数一样去传递查询语句
      • 例如

GET /cars/transactions/_validate/query?explain
{
“query”: {
“match”: {
“make”: “honda”
}
}
}

GET /cars/transactions/_search
{
“query”: {
“match_all”: {}
}
}

GET /cars/transactions/_search
{
“query”: {
“match_all”: {}
},
“_source”: [
“product_name”,
“price”
]
}

GET /cars/transactions/_search
{
“query”: {
“match_all”: {}
},
“sort”: [
{
“price”: “desc” ## 降序
}
]
}

GET /cars/transactions/_search
{
“query”: {
“match_all”: {}
},
“from”: 0, ## 从第几个开始查，最开始是 0
“size”: 1 ## 要查几个结果
}

GET /cars/transactions/_search
{
“query”: {
“range”: {
“price”: {
“gte”: 30000.00 ## 大于或等于30000
}
}
}
}

GET /cars/transactions/_search
{
“query”: {
“term”: {
“color”: “red” ## 查询的field和关键字
}
}
}

PUT my_index
{
  "mappings": {
    "my_type": {
      "properties": {
        "full_text": {
          "type":  "text"
        },
        "exact_value": {
          "type":  "keyword"
        }
      }
    }
  }
}

PUT my_index/my_type/1
{
  "full_text":   "Quick Foxes!",
  "exact_value": "Quick Foxes!"  
}


GET my_index/my_type/_search
{
  "query": {
    "term": {
      "exact_value": "Quick Foxes!"
    }
  }
}

// 可以请求出数据，因为exact_value字段没有被分析，可以完全匹配

GET my_index/my_type/_search
{
  "query": {
    "term": {
      "full_text": "Quick Foxes!"
    }
  }
}

// 请求不出数据，因为full_text分词后的结果中没有[Quick Foxes!]这个分词


PUT my_index
{
  "mappings": {
    "my_type": {
      "properties": {
        "full_text": {
          "type":  "text",
           "fields": {
            "keyword": { 
              "type": "keyword"
            }
          }
        },
        "exact_value": {
          "type":  "keyword"
        }
      }
    }
  }
}

GET /cars/transactions/_search
{
“query”: {
“match”: {
“make” : {
“query” : “bmw”
}
}
}
}

GET /cars/transactions/_search
{
“query”: {
“match”: {
“make” : {
“query” : “bmw honda”,
“operator”: “or”
}
}
}
}

GET /cars/transactions/_search
{
“query”: {
“match”: {
“make”: {
“query”: “bmw honda”,
“minimum_should_match”: “50%”
}
}
}
}

GET /cars/transactions/_search
{
“query”: {
“multi_match”: {
“query” : “red”,
“type”: “best_fields”,
“fields” : [“make”, “color”]
}
}
}

GET my_index/my_type/_search
{
“query”: {
“match”: {
“full_text”:{
“query”: “Quic”,
“fuzziness”: “AUTO”
}
}
}
}

GET /my_index/my_type/_search
{
“query”: {
“multi_match”: {
“fields”: [ “full_text” ],
“query”: “Quic”,
“fuzziness”: “AUTO”
}
}
}

GET my_index/my_type/_search
{
“query”: {
“match_phrase”: {
“exact_value” : {
“query” : “Quick Foxes!”
}
}
}
}

            Pos 1         Pos 2         Pos 3

Doc:        quick         brown         fox

Query:      quick         fox

Doc:        quick         brown         fox

Query:      fox           quick

GET my_index/my_type/_search
{
  "query": {
        "match_phrase": {
            "full_text" : {
                "query" : "foxes quick",
                "slop" : 3
            }
        }
  }
}

GET /cars/transactions/_search
{
“query”: {
“bool”: {
“must”: {
“term”: {
“color”: “red”
}
},
“must_not”: {
“term”: {
“make”: “bmw”
}
},
“filter”: {
“range”: {
“price”: {
“gt”: 10000
}
}
}
}
}
}
